=============VLAN-ы=============


nmcli connection add type bridge ifname hq-br con-name hq-br autoconnect yes bridge.vlan-filtering yes
- создание бриджа

type bridge — добавляет бридж
ifname hq-br — Будет создан виртуальный интерфейс с именем
con-name hq-br — имя профиля в NetworkManager

nmcli connection add type ethernet ifname ens224 con-name hq-br-slave autoconnect yes master hq-br
- созднмание "соединения"

nmcli connection add type vlan con-name {vlan№} ifname ens224.{vlan№} dev ens224 id {№} ipv4.addresses {ip адрес, например172.16.100.1/26} ipv4.method manual
- создание всех вланов

=============GRE=============


nmcli connection add type ip-tunnel con-name {Имя соединения} ifname {имя интерфейса, например gre1}
ip-tunnel.local {локальный ip - откуда стартует туннель (внешний на роутере)} ip-tunnel.remote {ip удалённого конца тоннеля}
ip-tunnel.mode gre ipv4.addresses 10.10.10.1/30 (30 - маска для 2х аддресов) ipv4.method manual
# создаёт профиль соединения типа "gre"

nmcli connection up {имя профиля} # включение тоннеля
